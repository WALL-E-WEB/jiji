# less递归

.wmixin(24); //声明 .wmixin为命名 而不是 class
.wmixin(@n, @i: 1) when (@i =< @n) { // if
  &.w-col-span-@{i} {
    width: @i / 24 * 100%;
  }
  .wmixin(@n, (@i + 1)); //递归
}

.generate-col(30);
.generate-col(@n, @i: 2) when (@i =< @n) {
  .p_@{i} {
    padding:@i
  }

  .p_l_@{i} {
    padding-left: @i;
  }
  .p_r_@{i} {
    padding-right: @i;
  }
  .p_b_@{i} {
    padding-bottom: @i;
  }
  .p_t_@{i} {
    padding-top: @i;
  }
  .generate-col(@n, (@i+2));
}

  }
  .zj_h_@{number} {
    height: ~'@{number}px';
  }

 .wh((@number - 2), @min);
